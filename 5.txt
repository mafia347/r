calculate_divisors <- function(n) {
  divisors <- c()
  
  for (i in 2:(n - 1)) {
    if (n %% i == 0) {
      divisors <- c(divisors, i)
    }
  }
  
  div_count <- length(divisors)
  
  if (div_count == 0) {
    message("The number ", n, " has no divisors (other than 1 and itself).")
  } else {
    cat("Divisors of", n, ":", divisors, "\n")
    cat("Number of divisors:", div_count, "\n")
  }
}


# Call the function with the integer 12
calculate_divisors(12)
